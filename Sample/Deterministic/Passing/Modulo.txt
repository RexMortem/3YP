// The modulo operator (%) and keyword form (mod) perform Euclidean remainder.
// Result is always non-negative for positive divisors.
output(10 % 3);    // 1
output(7 mod 2);   // 1
output(100 % 7);   // 2
output(0 % 5);     // 0

// Using modulo to check divisibility
fn is_divisible(n: int, d: int) -> bool {
    return n % d == 0;
}

output(is_divisible(12, 4));  // true
output(is_divisible(7, 3));   // false
