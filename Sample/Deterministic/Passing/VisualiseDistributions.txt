// :visualise() renders a horizontal bar chart of P(X = v) for each outcome.
// Works on any distribution; continuous distributions show min/max/mean instead.

// Fair six-sided die: uniform probability over {1,...,6}
let die = uniform(1, 6);
output(die:visualise());

// Biased coin: P(heads=1) = 0.7, P(tails=0) = 0.3
let coin = Discrete(0:0.3, 1:0.7);
output(coin:visualise());

// Continuous uniform: shows range and mean, not discrete bars
let continuous = uniformContinuous(0, 10);
output(continuous:visualise());
